Git Bash commands

clear // clears screen

git config user.name // shows user name
git config user.email // shows user email

git config --global user.name "Norm Straker" //configs user name
git config --global user.email normstraker@yahoo.com //configs user email

ls // list contents of current directory
ls -a // lists hidden files as well (hidden files begin with a .)
start . // opens file explorer of current directory
ls Directory name/ // lists contents of named directory
pwd // lists directory path of current location
dir // list files/folders in current directory
cd /directory path // change directory location specifically
cd directory name // change directory forward
cd .. // change directory backward (space between cd and ..)
cd . // list current directory and path


touch filename.ext // create file in current directory
touch filename.ext filename.ext filename.ext // create multiple files in current directory
mkdir name name name // creates directory or directories in current folder

rm filename.ext filename.ext // deletes the file/s permanently (not into the recycle bin)
rm -rf foldername // removes folders permanently
